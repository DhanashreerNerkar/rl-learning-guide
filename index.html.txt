    <script>
        // Lecture navigation function
        function openLecture(lectureUrl) {
            // Show loading toast
            showToast('Loading lecture... üìö', 'info');
            
            // Add small delay for UX
            setTimeout(() => {
                window.open(lectureUrl, '_blank');
                showToast('Lecture opened in new tab! üéâ', 'success');
            }, 500);
        }

        // Coming soon notification
        function showComingSoon() {
            showToast('Thanks for your interest! We\'ll notify you when Lecture 2 is ready üîî', 'info');
        }

        // Toast notification system
        function showToast(message, type = 'info') {
            const toast = document.getElementById('toast');
            toast.innerHTML = message;
            
            // Add type-specific styling
            toast.className = 'toast show';
            if (type === 'success') {
                toast.style.background = 'linear-gradient(45deg, #43e97b, #38f9d7)';
            } else if (type === 'info') {
                toast.style.background = 'linear-gradient(45deg, #667eea, #764ba2)';
            }
            
            // Auto hide after 3 seconds
            setTimeout(() => {
                toast.classList.remove('show');
            }, 3000);
        }

        // Add enhanced hover effects
        function addLectureEffects() {
            const availableCards = document.querySelectorAll('.lecture-card.available');
            
            availableCards.forEach(card => {
                card.addEventListener('mouseenter', function() {
                    this.style.transform = 'translateY(-8px) rotateX(2deg)';
                    this.style.transition = 'all 0.4s cubic-bezier(0.25, 0.46, 0.45, 0.94)';
                });
                
                card.addEventListener('mouseleave', function() {
                    this.style.transform = 'translateY(0) rotateX(0)';
                });
            });
        }

        // Animate progress bar
        function animateProgressBar() {
            const progressBar = document.querySelector('.progress-bar');
            
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.style.width = '50%';
                        entry.target.style.transition = 'width 2s ease-in-out';
                    }
                });
            });

            observer.observe(progressBar);
        }

        // Initialize page
        document.addEventListener('DOMContentLoaded', function() {
            // Add welcome message
            setTimeout(() => {
                showToast('Welcome to RL Course! Ready to learn? üéì', 'success');
            }, 1000);
            
            // Initialize animations and effects
            addLectureEffects();
            animateProgressBar();
            
            // Add keyboard navigation
            document.addEventListener('keydown', function(e) {
                if (e.key === '1' && e.ctrlKey) {
                    e.preventDefault();
                    openLecture('rl-visual-notes-01.html');
                }
            });
        });
